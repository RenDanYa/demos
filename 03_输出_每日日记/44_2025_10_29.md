## 时间统计

```dataviewjs
// 获取当前文件内容
let content = await dv.io.load(dv.current().file.path);
let lines = content.split('\n');

// 存储各标签时长（分钟）
let tagDurations = {};

// 解析时间记录
for (let line of lines) {
    let timeMatch = line.match(/(\d{2}:\d{2}) - (\d{2}:\d{2}) (.+?) #(.+)/);
    if (timeMatch) {
        let startTime = timeMatch[1];
        let endTime = timeMatch[2];
        let activity = timeMatch[3];
        let tags = timeMatch[4].split('/').filter(tag => tag.trim() !== '时间记录');
        
        // 计算时长（分钟）
        let start = new Date(`2000-01-01 ${startTime}`);
        let end = new Date(`2000-01-01 ${endTime}`);
        let duration = (end - start) / (1000 * 60);
        
        // 为每个标签累加时长
        tags.forEach(tag => {
            if (!tagDurations[tag]) {
                tagDurations[tag] = 0;
            }
            tagDurations[tag] += duration;
        });
    }
}

// 转换为小时和分钟格式
function formatDuration(minutes) {
    let hours = Math.floor(minutes / 60);
    let mins = Math.round(minutes % 60);
    return `${hours}小时${mins > 0 ? mins + '分钟' : ''}`;
}

// 创建表格显示结果
let tableData = Object.entries(tagDurations)
    .sort((a, b) => b[1] - a[1])
    .map(([tag, minutes]) => [tag, formatDuration(minutes), `${Math.round(minutes)}分钟`]);

dv.table(["标签", "总时长", "分钟数"], tableData);
```
