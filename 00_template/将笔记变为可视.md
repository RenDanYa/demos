<%*
// 第一步：清空当前文件内容
const currentFile = app.workspace.getActiveFile();
if (!currentFile) {
    new Notice("❌ 无法获取当前文件");
    return;
}

// 清空当前文件
await app.vault.modify(currentFile, "");
new Notice("✅ 页面内容已清空");

// 第二步：让用户选择要处理的文件
const files = app.vault.getMarkdownFiles();
const fileNames = files.map(f => f.path);
const selectedFile = await tp.system.suggester(fileNames, files, false, "请选择要处理的文件");

if (!selectedFile) {
    new Notice("已取消操作");
    return;
}

// 读取指定文件内容
const content = await app.vault.read(selectedFile);
const targetHeading = "## 💡核心思考与复盘";
const lines = content.split('\n');

let inTargetSection = false;
let extractedContent = [];
let foundSection = false;

// 查找并提取目标区域内容
for (let i = 0; i < lines.length; i++) {
    const line = lines[i].trim();
    
    // 找到目标标题
    if (line.startsWith(targetHeading)) {
        inTargetSection = true;
        foundSection = true;
        // 是否包含标题本身？这里我们包含
        extractedContent.push(lines[i]);
        continue;
    }
    
    // 如果在目标区域中
    if (inTargetSection) {
        // 如果遇到新的标题（相同或更高级别），则退出目标区域
        if (line.startsWith('#') && !line.startsWith('###')) {
            inTargetSection = false;
            continue;
        }
        
        // 如果是空行，跳过（或者保留）
        if (line === "") {
            extractedContent.push(lines[i]);
            continue;
        }
        
        // 匹配并转换 - [[...]] 格式的链接
        const match = lines[i].match(/^(\s*-\s*)\[\[([^#]+)#([^\]]+)\]\]/);
        if (match) {
            // 构建新的链接格式
            extractedContent.push(`${match[1]}![[${match[2]}#${match[3]}]]`);
        } else {
            // 如果不是目标格式的链接，保留原样
            extractedContent.push(lines[i]);
        }
    }
}

if (!foundSection) {
    new Notice(`在文件 ${selectedFile.name} 中未找到目标标题：${targetHeading}`);
    return;
}

// 将处理后的内容插入到当前光标位置（此时文件已清空，光标在文件开头）
if (extractedContent.length > 0) {
    // 添加一个分隔行和来源信息
    extractedContent.unshift(`\n---\n> 来源: [[${selectedFile.name}]]\n`);
    
    // 插入到当前文件开头
    await app.vault.append(currentFile, extractedContent.join('\n'));
    new Notice(`✅ 已从 ${selectedFile.name} 提取并转换内容到当前文件`);
}
%>
