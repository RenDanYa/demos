<%*
try {
    // 配置参数
    const targetHeading = "## 🛏️每日整理仪式";
    const targetFilePath = "resource/components/md/灵感.md"; // 请修改为实际路径

    // 方法1: 使用 active file
    let currentFile = app.workspace.getActiveFile();
    
    // 方法2: 如果方法1失败，尝试其他方式
    if (!currentFile) {
        // 通过当前窗口获取
        const activeLeaf = app.workspace.activeLeaf;
        if (activeLeaf) {
            currentFile = activeLeaf.view.file;
        }
    }
    
    // 方法3: 如果还是失败，使用 Templater 的替代方法
    if (!currentFile && tp && tp.file) {
        const allFiles = app.vault.getMarkdownFiles();
        currentFile = allFiles.find(f => f.path === tp.file.path);
    }

    if (!currentFile) {
        throw new Error("无法获取当前文件，请确保文件已保存且处于活动状态");
    }

    console.log("当前文件:", currentFile.path);

    // 读取文件内容
    let content = await app.vault.read(currentFile);
    
    // 查找目标标题
    const lines = content.split('\n');
    let startIndex = -1;
    let endIndex = lines.length;
    let found = false;

    for (let i = 0; i < lines.length; i++) {
        const line = lines[i].trim();
        
        if (line === targetHeading) {
            startIndex = i + 1;
            found = true;
            continue;
        }
        
        // 如果找到目标标题，查找下一个同级标题
        if (found && line.startsWith('## ') && !line.startsWith('###')) {
            endIndex = i;
            break;
        }
    }

    if (!found) {
        new Notice(`未找到标题: ${targetHeading}`);
        return;
    }

    // 提取内容
    const targetContent = lines.slice(startIndex, endIndex).join('\n').trim();
    
    if (!targetContent) {
        new Notice("目标标题下没有内容");
        return;
    }

    // 从原文件删除内容
    const newLines = [
        ...lines.slice(0, startIndex),
        ...lines.slice(endIndex)
    ];
    
    await app.vault.modify(currentFile, newLines.join('\n'));

    // 处理目标文件
    let targetFile = app.vault.getMarkdownFiles().find(f => f.path === targetFilePath);
    if (!targetFile) {
        targetFile = await app.vault.create(targetFilePath, `# 每日整理记录\n\n创建于: ${new Date().toLocaleString('zh-CN')}\n\n---\n\n`);
    }

    let targetFileContent = await app.vault.read(targetFile);
    const timestamp = new Date().toLocaleString('zh-CN');
    const newEntry = `## ${timestamp}的整理记录\n${targetContent}\n\n---\n\n`;

    // 添加到目标文件
    await app.vault.modify(targetFile, targetFileContent + newEntry);

    new Notice(`✅ 成功将内容移动到 ${targetFilePath}`);

} catch (error) {
    console.error("错误详情:", error);
    new Notice(`❌ 操作失败: ${error.message}`);
}
%>
