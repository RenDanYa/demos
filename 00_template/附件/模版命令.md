### 互动
<%*
// 1. 输入文本
let topic = await tp.system.prompt("请输入主题/标题")
let tags = await tp.system.prompt("请输入标签（用逗号分隔）", "idea, draft")

// 2. 从下拉列表中选择
let projectType = await tp.system.suggester(["会议记录", "读书笔记", "项目规划", "灵感随笔"], ["meeting", "book", "project", "idea"])
let priority = await tp.system.suggester(["高", "中", "低"], ["high", "medium", "low"])

// 3. 输入多行文本（用于摘要、内容等）
let summary = await tp.system.prompt("输入内容摘要", "这是关于...", true)
%>

# <% topic %>
**类型**: <% projectType %>
**优先级**: <% priority %>
**标签**: <% tags %>

## 摘要
<% summary %>

### 创建文件

<%*
try {
    // 1. 准备参数
    let fileName = "我的新笔记.md";
    let fileContent = `---
tags: [new]
created: tp.date.now()
---
# 这是我的新笔记
`;
    let openInNewTab = false;

    // 2. 执行创建
    let newFile = await tp.file.create_new(fileContent, fileName, openInNewTab);

    // 3. 检查结果
    if (newFile) {
        tR += "文件创建成功！链接：[[" + newFile.basename + "]]";
    } else {
        tR += "文件创建失败，返回了空值。";
    }

} catch (error) {
    // 4. 捕获并显示异常
    tR += "**发生错误：** " + error.message;
    console.error("Templater 错误: ", error); // 在开发者控制台（Ctrl+Shift+I）输出详细错误
}
%>
