<%*
// 通用修改front matter模板
const currentFile = app.workspace.getActiveFile();
if (currentFile) {
    const content = await app.vault.read(currentFile);
    const today = tp.date.now("YYYY-MM-DD");
    
    // 匹配front matter
    const frontMatterMatch = content.match(/^---\s*\n([\s\S]*?)\n---/);
    
    if (frontMatterMatch) {
        let frontMatter = frontMatterMatch[1];
        
        // 修改status和doneTime
        frontMatter = frontMatter.replace(/^status: .*$/m, 'status: DONE');
        
        // 处理doneTime，如果存在则更新，不存在则添加
        if (frontMatter.includes('doneTime:')) {
            frontMatter = frontMatter.replace(/^doneTime:.*$/m, `doneTime: ${today}`);
        } else {
            // 找到最后一行并在其后添加doneTime
            const lines = frontMatter.split('\n');
            lines.push(`doneTime: ${today}`);
            frontMatter = lines.join('\n');
        }
        
        // 重建文件内容
        const updatedContent = content.replace(/^---\s*\n[\s\S]*?\n---/, `---\n${frontMatter}\n---`);
        
        // 写回文件
        await app.vault.modify(currentFile, updatedContent);
        new Notice("✅ Front matter已更新，完成时间：" + today);
    } else {
        new Notice("❌ 未找到front matter");
    }
} else {
    new Notice("❌ 无法获取当前文件");
}
%>
