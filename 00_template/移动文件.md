<%*
try {
    const currentFile = app.workspace.getActiveFile();
    if (!currentFile) {
        new Notice("没有活动的文件");
        return;
    }

    // 使用 app.vault.getFolders() 方法（如果可用）
    let folders = [];
    
    // 尝试不同的方法获取文件夹列表
    if (app.vault.getFolders) {
        // Obsidian 1.0+ 的方法
        folders = app.vault.getFolders().map(folder => folder.path);
    } else {
        // 回退方法
        const allFiles = app.vault.getAllLoadedFiles();
        for (const file of allFiles) {
            if (file.children !== undefined) { // 这是文件夹
                folders.push(file.path === "" ? "/" : file.path);
            }
        }
    }
    
    // 过滤掉当前文件所在的文件夹
    const currentFolderPath = currentFile.parent ? (currentFile.parent.path === "" ? "/" : currentFile.parent.path) : "";
    folders = folders.filter(path => path !== currentFolderPath);
    
    if (folders.length === 0) {
        new Notice("没有其他文件夹可用");
        return;
    }
    
    folders.sort();
    
    // 添加新建文件夹选项
    const options = ["新建文件夹...", ...folders];
    
    const selected = await tp.system.suggester(options, options, true, "选择目标文件夹");
    
    if (!selected) return;
    
    let targetPath;
    
    if (selected === "新建文件夹...") {
        const newFolder = await tp.system.prompt("输入新文件夹路径:", "归档/" + tp.date.now("YYYY-MM-DD"));
        if (!newFolder) return;
        targetPath = newFolder;
    } else {
        targetPath = selected === "/" ? "" : selected;
    }
    
    // 确保目标文件夹存在
    const targetFolder = app.vault.getAbstractFileByPath(targetPath);
    if (!targetFolder) {
        await app.vault.createFolder(targetPath);
    }
    
    // 移动文件
    const newFilePath = targetPath + (targetPath ? "/" : "") + currentFile.name;
    await app.fileManager.renameFile(currentFile, newFilePath);
    
    new Notice(`文件已移动到: ${newFilePath}`);
    
} catch (error) {
    new Notice(`移动文件时出错: ${error.message}`);
    console.error("Templater 文件移动错误:", error);
}
%>
