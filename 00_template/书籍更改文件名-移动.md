<%*
// ============================================
// 静默文件复制工具 - 目标文件夹: inbox
// ============================================

try {
    // 获取当前活动文件
    let currentFile = app.workspace.getActiveFile()
    
    if (!currentFile) {
        // 静默失败，不显示任何信息
        return
    }

    // 获取新文件名
    let defaultNewName = currentFile.basename + " - 副本"
    let newName = await tp.system.prompt(
        "输入副本文件名",
        defaultNewName
    )
    
    if (!newName) {
        // 静默取消
        return
    }

    // 确保文件名有扩展名
    if (!newName.endsWith('.md')) {
        newName += '.md'
    }

    // 固定目标文件夹为 inbox
    let targetFolder = "inbox"
    
    // 构建目标路径
    let targetPath = `${targetFolder}/${newName}`

    // 确保 inbox 文件夹存在
    let inboxFolder = app.vault.getAbstractFileByPath(targetFolder)
    if (!inboxFolder) {
        await app.vault.createFolder(targetFolder)
    }

    // 检查文件是否已存在，如果存在则自动添加序号
    let finalTargetPath = targetPath
    let counter = 1
    while (app.vault.getAbstractFileByPath(finalTargetPath)) {
        let nameWithoutExt = newName.replace('.md', '')
        finalTargetPath = `${targetFolder}/${nameWithoutExt} (${counter}).md`
        counter++
    }

    // 读取源文件内容并创建副本
    let fileContent = await app.vault.read(currentFile)
    await app.vault.create(finalTargetPath, fileContent)
    
    // 静默成功，不显示任何信息

} catch (error) {
    // 静默失败，不显示任何信息
}
%>
